$("#unit").keypress(function(e) {
    var charCode = (e.which) ? e.which : e.keyCode;
    if (charCode > 31 && (charCode < 48 || charCode > 57)) {
        return false;
    }
    if ($("#unit").val().length >= 9) {
        return false;
    }
    return true;
});
